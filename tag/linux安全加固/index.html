root:x:0:0:root:/root:/bin/bash
[root@localhost ~]# passwd -l XXXXX^C	#锁定账户,建议如果存在多个UID为0用户,先锁定再去排查,最后在确定是否删除或解除

[root@localhost ~]# passwd -u XXXXX^	#解除锁定
